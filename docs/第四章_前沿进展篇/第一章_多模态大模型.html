卷积神经网络（CNN）：

CNN长期以来是计算机视觉的主导架构。

import torch
import torch.nn as nn

class SimpleCNN(nn.Module):
    """简单CNN模型"""
    def __init__(self, num_classes=10):
        super().__init__()
        self.features = nn.Sequential(
            # 卷积层1
            nn.Conv2d(3, 64, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),
            # 卷积层2
            nn.Conv2d(64, 128, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),
            # 卷积层3
            nn.Conv2d(128, 256, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2)
        )
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(256 * 4 * 4, 512),
            nn.ReLU(),
            nn.Linear(512, num_classes)
        )

    def forward(self, x):
        x = self.features(x)
        x = self.classifier(x)
        return x

Vision Transformer (ViT)：

ViT将Transformer架构引入视觉领域。

class VisionTransformer(nn.Module):
    """Vision Transformer"""
    def __init__(self, image_size=224, patch_size=16, num_classes=1000,
                 dim=768, depth=12, heads=12, mlp_ratio=4):
        super().__init__()
        self.patch_size = patch_size
        num_patches = (image_size // patch_size) ** 2

        # Patch嵌入
        self.patch_embed = nn.Conv2d(3, dim, kernel_size=patch_size, stride=patch_size)

        # 类别token和位置编码
        self.cls_token = nn.Parameter(torch.randn(1, 1, dim))
        self.pos_embed = nn.Parameter(torch.randn(1, num_patches + 1, dim))

        # Transformer编码器
        self.blocks = nn.ModuleList([
            TransformerBlock(dim, heads, mlp_ratio)
            for _ in range(depth)
        ])

        self.norm = nn.LayerNorm(dim)
        self.head = nn.Linear(dim, num_classes)

    def forward(self, x):
        B = x.shape[0]

        # 分割为patches
        x = self.patch_embed(x)  # (B, dim, H/P, W/P)
        x = x.flatten(2).transpose(1, 2)  # (B, N, dim)

        # 添加cls token
        cls_tokens = self.cls_token.expand(B, -1, -1)
        x = torch.cat([cls_tokens, x], dim=1)

        # 添加位置编码
        x = x + self.pos_embed

        # Transformer编码
        for block in self.blocks:
            x = block(x)

        x = self.norm(x)

        # 分类
        return self.head(x[:, 0])

class TransformerBlock(nn.Module):
    """Transformer块"""
    def __init__(self, dim, heads, mlp_ratio=4):
        super().__init__()
        self.norm1 = nn.LayerNorm(dim)
        self.attn = nn.MultiheadAttention(dim, heads)
        self.norm2 = nn.LayerNorm(dim)
        self.mlp = nn.Sequential(
            nn.Linear(dim, dim * mlp_ratio),
            nn.GELU(),
            nn.Linear(dim * mlp_ratio, dim)
        )

    def forward(self, x):
        x = x + self.attn(self.norm1(x), self.norm1(x), self.norm1(x))[0]
        x = x + self.mlp(self.norm2(x))
        return x

import librosa
import torch
import torchaudio

class AudioFeatureExtractor:
    """音频特征提取器"""
    def __init__(self, sample_rate=16000, n_mels=80, n_fft=512):
        self.sample_rate = sample_rate
        self.n_mels = n_mels
        self.n_fft = n_fft

        # Mel频谱变换
        self.mel_transform = torchaudio.transforms.MelSpectrogram(
            sample_rate=sample_rate,
            n_fft=n_fft,
            n_mels=n_mels
        )

    def extract_mel_spectrogram(self, audio_path):
        """提取Mel频谱"""
        waveform, sr = torchaudio.load(audio_path)

        # 重采样
        if sr != self.sample_rate:
            resampler = torchaudio.transforms.Resample(sr, self.sample_rate)
            waveform = resampler(waveform)

        # 提取Mel频谱
        mel_spec = self.mel_transform(waveform)
        mel_spec_db = torchaudio.transforms.AmplitudeToDB()(mel_spec)

        return mel_spec_db

    def extract_mfcc(self, audio_path, num_mfcc=40):
        """提取MFCC特征"""
        waveform, sr = librosa.load(audio_path, sr=self.sample_rate)

        # MFCC
        mfcc = librosa.feature.mfcc(
            y=waveform,
            sr=sr,
            n_mfcc=num_mfcc
        )

        return torch.from_numpy(mfcc)

CLIP对比学习：

CLIP通过对比学习对齐图像和文本。

class CLIPModel(nn.Module):
    """CLIP模型"""
    def __init__(self, image_encoder, text_encoder, dim=512):
        super().__init__()
        self.image_encoder = image_encoder
        self.text_encoder = text_encoder
        self.dim = dim

        # 投影层
        self.image_proj = nn.Linear(dim, dim)
        self.text_proj = nn.Linear(dim, dim)

    def encode_image(self, image):
        """编码图像"""
        features = self.image_encoder(image)
        features = self.image_proj(features)
        return features / features.norm(dim=-1, keepdim=True)

    def encode_text(self, text):
        """编码文本"""
        features = self.text_encoder(text)
        features = self.text_proj(features)
        return features / features.norm(dim=-1, keepdim=True)

    def forward(self, images, texts):
        """对比学习前向"""
        image_features = self.encode_image(images)
        text_features = self.encode_text(texts)

        # 计算相似度
        logits = image_features @ text_features.T / 0.07

        # 标签：对角线为正样本
        batch_size = images.shape[0]
        labels = torch.arange(batch_size, device=images.device)

        # 图像到文本和文本到图像的损失
        loss_i2t = nn.functional.cross_entropy(logits, labels)
        loss_t2i = nn.functional.cross_entropy(logits.T, labels)
        loss = (loss_i2t + loss_t2i) / 2

        return loss, logits

1.2 多模态模型架构#

1.2.1 CLIP模型详解#

CLIP（Contrastive Language-Image Pre-training）是多模态学习的里程碑。

模型架构：

class CLIP(nn.Module):
    """CLIP模型实现"""
    def __init__(self, image_dim=768, text_dim=768, embed_dim=512):
        super().__init__()

        # 图像编码器（可用ViT或ResNet）
        self.image_encoder = VisionTransformer(
            dim=image_dim,
            num_patches=256,
            depth=12
        )

        # 文本编码器（Transformer）
        self.text_encoder = TextTransformer(
            dim=text_dim,
            num_tokens=49408,
            depth=12
        )

        # 投影
        self.image_projection = nn.Linear(image_dim, embed_dim)
        self.text_projection = nn.Linear(text_dim, embed_dim)

        # 温度参数
        self.logit_scale = nn.Parameter(torch.ones([]) * 2.659)

    def encode_image(self, image):
        x = self.image_encoder(image)
        x = self.image_projection(x)
        return x / x.norm(dim=-1, keepdim=True)

    def encode_text(self, text):
        x = self.text_encoder(text)
        x = self.text_projection(x)
        return x / x.norm(dim=-1, keepdim=True)

    def forward(self, images, text_tokens):
        # 编码
        image_features = self.encode_image(images)
        text_features = self.encode_text(text_tokens)

        # 对比损失
        logits = image_features @ text_features.T * self.logit_scale.exp()

        return logits

预训练过程：

class CLIPPretraining:
    """CLIP预训练"""
    def __init__(self, model, device='cuda'):
        self.model = model
        self.device = device

    def contrastive_loss(self, image_features, text_features, batch_size):
        """对比损失"""
        # 温度缩放
        logit_scale = self.model.logit_scale.exp()

        # 相似度矩阵
        logits = image_features @ text_features.T * logit_scale

        # 标签
        labels = torch.arange(batch_size, device=self.device)

        # 对称损失
        loss_image = nn.functional.cross_entropy(logits, labels)
        loss_text = nn.functional.cross_entropy(logits.T, labels)

        return (loss_image + loss_text) / 2

    def train(self, dataloader, epochs=32):
        """预训练"""
        optimizer = torch.optim.AdamW(self.model.parameters(), lr=5e-4)

        for epoch in range(epochs):
            for batch in dataloader:
                images = batch['image'].to(self.device)
                text_tokens = batch['text_tokens'].to(self.device)

                # 前向
                image_features = self.model.encode_image(images)
                text_features = self.model.encode_text(text_tokens)

                # 损失
                loss = self.contrastive_loss(image_features, text_features, images.size(0))

                # 反向
                optimizer.zero_grad()
                loss.backward()
                optimizer.step()

            print(f"Epoch {epoch}, Loss: {loss.item():.4f}")

1.2.2 DALL-E、Midjourney#

DALL-E架构：

DALL-E使用dVAE + CLIP的架构生成图像。

class DALL_E(nn.Module):
    """DALL-E简化版"""
    def __init__(self, image_tokens=1024, text_dim=256, dim=64):
        super().__init__()
        self.image_tokens = image_tokens

        # dVAE（离散VAE用于图像）
        self.dvae = dVAE()

        # CLIP文本编码器
        self.text_encoder = CLIPTextEncoder(dim=text_dim)

        # 生成Transformer
        self.transformer = TransformerDecoder(
            dim=dim,
            num_tokens=8192,  # 8192个离散token
            depth=64,
            heads=16
        )

    def generate(self, text_prompt, num_images=1):
        """从文本生成图像"""
        # 编码文本
        text_features = self.text_encoder(text_prompt)

        # 自回归生成离散token
        tokens = []
        for _ in range(self.image_tokens):
            logits = self.transformer(text_features, tokens)
            token = torch.argmax(logits[:, -1], dim=-1)
            tokens.append(token)

        # dVAE解码
        image_tokens = torch.stack(tokens, dim=1)
        images = self.dvae.decode(image_tokens)

        return images

1.2.4 Flamingo等模型#

Flamingo是DeepMind提出的少样本视觉语言模型。

class FlamingoModel(nn.Module):
    """Flamingo架构"""
    def __init__(self, vision_encoder, language_model, dim=768):
        super().__init__()
        self.vision_encoder = vision_encoder
        self.language_model = language_model

        # Perceiver Resampler（融合视觉特征）
        self.perceiver = PerceiverResampler(dim=dim)

        # 门控交叉注意力
        self.gated_cross_attention = GatedCrossAttention(dim=dim)

    def forward(self, images, text, past_key_values=None):
        """Flamingo前向"""
        # 编码视觉特征
        visual_features = self.vision_encoder(images)
        visual_features = self.perceiver(visual_features)

        # 门控交叉注意力
        if past_key_values is None:
            past_key_values = {}

        outputs = self.gated_cross_attention(
            text,
            visual_features,
            past_key_values
        )

        return outputs

class PerceiverResampler(nn.Module):
    """Perceiver Resampler - 将变长视觉特征转为固定长度"""
    def __init__(self, dim=768, num_latents=64):
        super().__init__()
        self.num_latents = num_latents

        # 可学习的查询向量
        self.latents = nn.Parameter(torch.randn(1, num_latents, dim))

        # 交叉注意力
        self.cross_attention = nn.MultiheadAttention(dim, heads=8)

    def forward(self, visual_features):
        """
        visual_features: (batch, num_patches, visual_dim)
        """
        B = visual_features.shape[0]
        latents = self.latents.expand(B, -1, -1)

        # 交叉注意力
        for _ in range(3):
            latents = self.cross_attention(
                latents, visual_features, visual_features
            )[0] + latents

        return latents

from diffusers import StableDiffusionPipeline
import torch

class TextToImageGenerator:
    """文本到图像生成器"""
    def __init__(self, model_id="stabilityai/stable-diffusion-2-1"):
        self.pipe = StableDiffusionPipeline.from_pretrained(
            model_id,
            torch_dtype=torch.float16,
            device_map="auto"
        )

    def generate(self, prompt, num_images=1, guidance_scale=7.5):
        """生成图像"""
        images = self.pipe(
            prompt,
            num_images_per_prompt=num_images,
            guidance_scale=guidance_scale,
            num_inference_steps=50
        ).images

        return images

    def generate_variations(self, image, prompt=None):
        """从图像生成变体"""
        if prompt:
            images = self.pipe(prompt, image=image).images
        else:
            images = self.pipe(image=image).images
        return images

class ImageCaptioning:
    """图像描述生成"""
    def __init__(self, model_name="Salesforce/blip-image-captioning-base"):
        from transformers import BlipProcessor, BlipForConditionalGeneration

        self.processor = BlipProcessor.from_pretrained(model_name)
        self.model = BlipForConditionalGeneration.from_pretrained(model_name)

    def caption(self, image_path, prompt=None):
        """生成图像描述"""
        from PIL import Image

        image = Image.open(image_path)

        if prompt:
            inputs = self.processor(image, prompt, return_tensors="pt")
        else:
            inputs = self.processor(image, return_tensors="pt")

        output = self.model.generate(**inputs, max_new_tokens=200)
        caption = self.processor.decode(output[0], skip_special_tokens=True)

        return caption

    def describe_detailed(self, image_path):
        """详细图像分析"""
        prompt = "请详细描述这张图片，包括场景、物体、颜色等细节"
        return self.caption(image_path, prompt)

1.3.3 语音交互#

语音交互实现人机自然对话。

class VoiceAssistant:
    """语音助手"""
    def __init__(self, model_name="meta-llama/Llama-2-7b-chat-hf"):
        # 语音识别
        self.asr = self._load_asr()

        # 语言模型
        self.llm = self._load_llm(model_name)

        # 语音合成
        self.tts = self._load_tts()

    def _load_asr(self):
        """加载语音识别模型"""
        from transformers import WhisperProcessor, WhisperForConditionalGeneration

        processor = WhisperProcessor.from_pretrained("openai/whisper-medium")
        model = WhisperForConditionalGeneration.from_pretrained("openai/whisper-medium")
        return {"processor": processor, "model": model}

    def _load_tts(self):
        """加载语音合成模型"""
        # 使用Coqui TTS
        from TTS.api import TTS
        return TTS("coqui/XTTS-v2")

    def listen(self, audio_path):
        """语音转文字"""
        import librosa

        audio, sr = librosa.load(audio_path, sr=16000)

        inputs = self.asr["processor"](audio, return_tensors="pt", sampling_rate=16000)
        forced_decoder_ids = self.asr["processor"].get_decoder_prompt_ids(language="zh", task="transcribe")

        output = self.asr["model"].generate(
            inputs["input_features"],
            forced_decoder_ids=forced_decoder_ids
        )

        text = self.asr["processor"].decode(output[0], skip_special_tokens=True)
        return text

    def speak(self, text, output_path):
        """文字转语音"""
        self.tts.tts_to_file(text=text, file_path=output_path)

    def chat(self, audio_input):
        """语音对话"""
        # 语音转文字
        user_text = self.listen(audio_input)

        # 对话
        response = self.llm.generate(user_text)

        # 文字转语音
        audio_output = "response.wav"
        self.speak(response, audio_output)

        return response, audio_output

class VideoUnderstanding:
    """视频理解模型"""
    def __init__(self):
        self.frame_sampler = UniformFrameSampler(num_frames=16)

    def extract_features(self, video_path):
        """提取视频特征"""
        import decord

        vr = decord.VideoReader(video_path)
        frames = vr.get_batch(range(len(vr)))

        # 采样帧
        frame_indices = self.frame_sampler.sample(len(vr))
        sampled_frames = frames[frame_indices].asnumpy()

        # 逐帧编码
        features = []
        for frame in sampled_frames:
            frame_tensor = torch.from_numpy(frame).permute(2, 0, 1).unsqueeze(0)
            feature = self.frame_encoder(frame_tensor)
            features.append(feature)

        return torch.stack(features)

    def understand_video(self, video_path, query):
        """视频问答"""
        # 提取视频特征
        video_features = self.extract_features(video_path)

        # 构建prompt
        prompt = f"视频中包含的内容：{self.describe_video(video_features)}。问题是：{query}"

        # 生成回答
        answer = self.llm.generate(prompt)

        return answer

    def describe_video(self, features):
        """视频描述"""
        # 聚合时空特征
        aggregated = features.mean(dim=0)  # 时间平均
        return aggregated

多模态融合正从弱融合走向强融合。

统一多模态模型：

class UnifiedMultimodalModel(nn.Module):
    """统一多模态模型"""
    def __init__(self, config):
        super().__init__()
        self.modalities = config.modalities  # ["text", "image", "audio", "video"]

        # 各模态的编码器
        self.encoders = nn.ModuleDict({
            mod: create_encoder(mod, config)
            for mod in self.modalities
        })

        # 投影到统一空间
        self.projection = nn.Linear(config.hidden_size, config.unified_dim)

        # Transformer融合
        self.fusion_transformer = TransformerEncoder(
            num_layers=config.num_layers,
            dim=config.unified_dim
        )

    def encode(self, modality, data):
        """编码单模态"""
        encoder = self.encoders[modality]
        features = encoder(data)
        return self.projection(features)

    def fuse(self, encoded_modalities):
        """融合多模态"""
        # 拼接所有模态
        fused = torch.cat(encoded_modalities.values(), dim=1)

        # Transformer融合
        return self.fusion_transformer(fused)

class RealTimeMultimodal:
    """实时多模态处理"""
    def __init__(self):
        self.frame_buffer = []
        self.buffer_size = 10

    def process_frame(self, frame):
        """处理单帧"""
        # 添加到缓冲区
        self.frame_buffer.append(frame)
        if len(self.frame_buffer) > self.buffer_size:
            self.frame_buffer.pop(0)

        # 实时特征提取
        features = self.extract_features(frame)

        # 快速处理
        result = self.quick_inference(features)

        return result

    async def continuous_processing(self, camera_stream):
        """连续处理摄像头流"""
        async for frame in camera_stream:
            result = self.process_frame(frame)

            # 实时显示结果
            self.display(result)

            # 非阻塞处理下一帧
            await asyncio.sleep(0.033)  # ~30fps
