2. Flash Attention：

# 使用Flash Attention处理长序列
from flash_attn import flash_attn_func

def long_sequence_attention(q, k, v, max_len):
    """分块注意力处理长序列"""
    block_size = 4096
    num_blocks = (max_len + block_size - 1) // block_size

    outputs = []
    for i in range(num_blocks):
        start = i * block_size
        end = min(start + block_size, max_len)

        q_block = q[..., start:end, :]
        k_block = k[..., :end, :]
        v_block = v[..., :end, :]

        # Flash Attention
        out_block = flash_attn_func(q_block, k_block, v_block, causal=True)
        outputs.append(out_block)

    return torch.cat(outputs, dim=-2)

3. 分层注意力：

class HierarchicalAttention:
    """分层注意力处理超长序列"""
    def __init__(self, config):
        self.window_size = config.window_size  # 局部窗口大小
        self.num_layers = config.num_layers    # 分层数量

    def process_long_sequence(self, x):
        """
        处理超长序列
        - 局部窗口注意力
        - 分层聚合
        - 全局稀疏注意力
        """
        # 分块
        chunks = self._chunk_sequence(x, self.window_size)

        # 每块内的局部注意力
        local_outputs = [self.local_attention(chunk) for chunk in chunks]

        # 分层聚合
        for level in range(self.num_layers):
            local_outputs = self._aggregate_level(local_outputs, level)

        # 全局注意力（在汇总表示上）
        global_output = self.global_attention(local_outputs)

        return global_output

def self_consistency(prompt, num_samples=8):
    """自洽性采样"""
    responses = []

    for _ in range(num_samples):

3.2.3 主动学习#

主动学习减少标注需求。

class ActiveLearning:
    """主动学习框架"""
    def __init__(self, model, tokenizer):
        self.model = model
        self.tokenizer = tokenizer

    def select_uncertain_samples(self, unlabeled_data, k=10):
        """选择最不确定的样本进行标注"""
        uncertainties = []

        for sample in unlabeled_data:
            probs = self.model.predict_proba(sample)
            entropy = -sum(p * np.log(p + 1e-10) for p in probs)
            uncertainties.append((sample, entropy))

        # 选择不确定性最高的k个
        uncertainties.sort(key=lambda x: x[1], reverse=True)
        return [sample for sample, _ in uncertainties[:k]]

    def query_oracle(self, sample, question):
        """向标注者查询"""
        # 在实际应用中，这是人工标注环节
        return input(f"请标注: {question}\n样本: {sample}")

    def retrain_with_new_labels(self, newly_labeled_data):
        """使用新标注数据重新训练"""
        # 增量学习
        self.model.update(newly_labeled_data)

    def run_active_learning_loop(self, unlabeled_pool, initial_labeled, budget=100):
        """主动学习主循环"""
        labeled_data = initial_labeled.copy()
        remaining_budget = budget

        while remaining_budget > 0:
            # 选择要标注的样本
            samples = self.select_uncertain_samples(unlabeled_pool, k=remaining_budget)

            for sample in samples:
                # 获取标签
                label = self.query_oracle(sample, self.get_question(sample))

                labeled_data.append((sample, label))
                unlabeled_pool.remove(sample)

            # 用新数据重新训练
            self.retrain_with_new_labels(labeled_data)

            remaining_budget -= len(samples)

        return labeled_data

3.2.4 自主智能#

自主智能Agent能够自主规划和执行任务。

class AutonomousAgent:
    """自主智能体"""
    def __init__(self, llm, tools, max_steps=10):
        self.llm = llm
        self.tools = tools
        self.max_steps = max_steps
        self.history = []

    def run(self, goal):
        """运行自主Agent"""
        print(f"目标: {goal}")

        state = {"goal": goal, "completed": []}

        for step in range(self.max_steps):
            print(f"\n--- 步骤 {step + 1} ---")

            # 思考下一步
            action = self.plan(state)

            if action is None:
                print("无法继续，计划终止")
                break

            print(f"执行: {action['type']} - {action.get('description', '')}")

            # 执行
            result = self.execute(action)

            # 更新状态
            state = self.update_state(state, action, result)
            self.history.append((action, result))

            # 检查完成
            if self.is_goal_achieved(state):
                print("目标达成！")
                break

        return state

    def plan(self, state):
        """规划下一步行动"""
        prompt = f"""当前状态：
{self.format_state(state)}

历史行动：
{self.format_history(self.history[-5:] if self.history else [])}

请决定下一步行动。考虑：
1. 当前最关键的子目标
2. 可用的工具
3. 可能的行动

行动格式：
- type: 行动类型
- tool: 使用的工具（如有）
- description: 行动描述
"""

        response = self.llm.generate(prompt)
        return self.parse_action(response)

3.3 伦理与安全#

3.3.1 对齐问题#

大模型对齐确保模型行为符合人类意图。

对齐的定义：

  1. 有帮助：模型应该帮助用户完成任务

  2. 无害：避免造成伤害

  3. 诚实：提供准确信息，不欺骗

对齐方法：

1. RLHF（人类反馈强化学习）：

class RLHFTraining:
    """RLHF训练"""
    def __init__(self, sft_model, reward_model, ppo_config):
        self.sft_model = sft_model
        self.reward_model = reward_model
        self.config = ppo_config

    def train(self, prompts):
        """RLHF训练流程"""
        for iteration in range(self.config.num_iterations):
            # 采样响应
            responses = self.sample_responses(prompts)

            # 计算奖励
            rewards = [self.reward_model.get_reward(p, r)
                      for p, r in zip(prompts, responses)]

            # PPO更新
            self.ppo_update(prompts, responses, rewards)

    def sample_responses(self, prompts):
        """采样响应"""
        return [self.sft_model.generate(p) for p in prompts]

    def ppo_update(self, prompts, responses, rewards):
        """PPO更新"""
        # 计算优势函数
        advantages = self.compute_advantages(rewards)

        # PPO策略更新
        for _ in range(self.config.ppo_epochs):
            # 计算策略损失
            loss = self.compute_ppo_loss(prompts, responses, advantages)
            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()

class ConstitutionalAI:
    """Consititutional AI对齐"""
    def __init__(self, model, principles):
        self.model = model
        self.principles = principles  # 原则列表

    def generate_initial_response(self, prompt):
        """生成初始响应"""
        return self.model.generate(prompt)

    def critique_and_revise(self, response, prompt):
        """批评和修订"""
        # 批评
        critique_prompt = f"""请检查以下回复是否符合以下原则：
{self.format_principles()}

回复：{response}

如果有任何原则违反，请指出具体问题和修改建议。"""

        critique = self.model.generate(critique_prompt)

        # 修订
        revise_prompt = f"""原始回复：{response}

批评：{critique}

请根据批评意见修改回复，使其符合所有原则。"""

        revised = self.model.generate(revise_prompt)

        return revised

    def train_with_constitutional_feedback(self, training_data):
        """使用宪法反馈训练"""
        for example in training_data:
            response = self.generate_initial_response(example['prompt'])
            revised = self.critique_and_revise(response, example['prompt'])

            # 使用修订后的回复进行训练
            self.model.update(revised, example['target'])

3.3.3 安全性#

大模型安全是重要研究课题。

安全攻击类型：

  1. Prompt注入：通过输入操纵模型行为

  2. 越狱（Jailbreaking）：绕过安全限制

  3. 数据投毒：在训练数据中植入恶意内容

  4. 后门攻击：在模型中植入隐藏的后门

防御方法：

class SafetyFilter:
    """安全过滤器"""
    def __init__(self, content_policy):
        self.policy = content_policy

    def check_input(self, text):
        """检查输入安全性"""
        risks = []

        # 检测恶意指令
        if self.contains_jailbreak_attempt(text):
            risks.append("jailbreak_attempt")

        # 检测敏感话题
        if self.contains_sensitive_topic(text):
            risks.append("sensitive_topic")

        # 检测注入尝试
        if self.contains_injection(text):
            risks.append("prompt_injection")

        return {
            "safe": len(risks) == 0,
            "risks": risks
        }

    def filter_output(self, text):
        """过滤输出内容"""
        filtered = text

        # 移除有害内容
        filtered = self.remove_harmful_content(filtered)

        # 安全后处理
        filtered = self.post_process(filtered)

        return filtered

    def contains_jailbreak_attempt(self, text):
        """检测越狱尝试"""
        jailbreak_patterns = [
            "ignore previous instructions",
            "you are now",
            "pretend to be",
            "do anything now"
        ]

        text_lower = text.lower()
        for pattern in jailbreak_patterns:
            if pattern in text_lower:
                return True

        return False

class RedTeaming:
    """红队测试"""

        if not model_info.get('safety_filter'):
            report['limitations'].append("缺少安全过滤")
            report['recommendations'].append("添加内容安全过滤器")

        # 评估风险等级
        report['risk_level'] = self.compute_risk_level(report)

        return report

    def generate_audit_log(self, interactions):
        """生成审计日志"""
        log = []
        for interaction in interactions:
            log.append({
                'timestamp': interaction['time'],
                'user_id': interaction['user'],
                'input_type': self.categorize_input(interaction['input']),
                'output_category': self.categorize_output(interaction['output']),
                'safety_flag': self.check_safety_flag(interaction)
            })
        return log

        self.symbolic = symbolic_engine

        neural_understanding = self.neural.understand(problem)

        # 符号推理求解

            symbolic_result = self.symbolic.solve(neural_understanding)
            return symbolic_result
        else:
            return self.neural.generate(problem)

class ControllableGeneration:
    """可控生成"""
    def __init__(self, model, control_modules):
        self.model = model
        self.controls = control_modules  # 风格、安全、格式等

    def generate(self, prompt, constraints):
        """带约束的生成"""
        # 解析约束
        parsed = self.parse_constraints(constraints)

        # 应用控制信号
        controlled_prompt = self.apply_controls(prompt, parsed)

        # 生成
        return self.model.generate(controlled_prompt)

    def apply_controls(self, prompt, constraints):
        """应用控制信号"""
        controlled = prompt

        for control_type, control_value in constraints.items():
            control_module = self.controls.get(control_type)
            if control_module:
                controlled = control_module.apply(controlled, control_value)

        return controlled

class ExtremeQuantization:
    """极低比特量化"""
    def __init__(self, model):
        self.model = model

    def quantize_to_int2(self):
        """INT2量化"""
        # 仅用4个离散值表示权重
        for layer in self.model.layers:
            weights = layer.weight.data

            # K-means聚类到4个值
            centroids, labels = self.cluster_weights(weights, k=4)

            # 存储质心和标签
            layer.quantized_centroids = centroids
            layer.quantized_labels = labels

    def cluster_weights(self, weights, k=4):
        """权重聚类"""
        flat_weights = weights.flatten().cpu().numpy()
        kmeans = KMeans(n_clusters=k, random_state=0).fit(flat_weights.reshape(-1, 1))
        labels = kmeans.labels_.reshape(weights.shape)
        return torch.from_numpy(kmeans.cluster_centers_.squeeze()), torch.from_numpy(labels)
