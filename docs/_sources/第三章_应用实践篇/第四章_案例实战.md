```python
from transformers import AutoTokenizer, AutoModelForCausalLM
from transformers import Trainer, TrainingArguments, DataCollatorForLanguageModeling

def fine_tune_model(model_name, train_dataset, eval_dataset, output_dir):
    """微调语言模型"""
    # 加载模型和分词器
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    model = AutoModelForCausalLM.from_pretrained(model_name)

    # 配置tokenizer
    tokenizer.pad_token = tokenizer.eos_token
    model.config.pad_token_id = tokenizer.pad_token_id

    # 数据整理器
    data_collator = DataCollatorForLanguageModeling(
        tokenizer=tokenizer,
        mlm=False  # Causal LM不使用MLM
    )

    # 训练参数
    training_args = TrainingArguments(
        output_dir=output_dir,
        num_train_epochs=3,
        per_device_train_batch_size=8,
        gradient_accumulation_steps=4,
        evaluation_strategy="epoch",
        save_strategy="epoch",
        logging_dir="./logs",
        fp16=True,
        optim="paged_adamw_32bit"
    )

    # Trainer
    trainer = Trainer(
        model=model,
        args=training_args,
        train_dataset=train_dataset,
        eval_dataset=eval_dataset,
        data_collator=data_collator
    )

    trainer.train()
    return model
```

def chat_with_llama(prompt, history=None):
    """与LLaMA对话"""
    if history:
        # 构建多轮对话
        conversation = ""
        for user, assistant in in history:
            conversation += f"用户: {user}\n助手: {assistant}\n"
        conversation += f"用户: {prompt}\n助手: "
    else:
        conversation = f"用户: {prompt}\n助手: "

    inputs = tokenizer(conversation, return_tensors="pt").to(model.device)
    outputs = model.generate(**inputs, max_new_tokens=256)
    response = tokenizer.decode(outputs[0], skip_special_tokens=True)

    return response.split("助手: ")[-1]

    """加载模型进行QLoRA微调"""
    # 加载模型（4bit量化）
    model = AutoModelForCausalLM.from_pretrained(
        model_name,

        torch_dtype=torch.float16,

    # 配置LoRA
    lora_config = LoraConfig(
        task_type=TaskType.CAUSAL_LM,
        r=64,
        lora_alpha=16,
        target_modules=["q_proj", "v_proj", "k_proj", "o_proj"],
        lora_dropout=0.05,
        bias="none"
    )

    # 应用LoRA
    model = get_peft_model(model, lora_config)
    model.print_trainable_parameters()

    return model

def train_with_qlora(model, train_dataset, eval_dataset):
    """QLoRA训练"""
    training_args = TrainingArguments(
        output_dir="./llama-qlora",
        num_train_epochs=3,
        per_device_train_batch_size=8,

        eval_steps=100,
        save_strategy="steps",
        logging_steps=10,
        learning_rate=2e-4,
        fp16=True,
        warmup_ratio=0.03
    )

    trainer = Trainer(
        model=model,
        args=training_args,
        train_dataset=train_dataset,
        eval_dataset=eval_dataset
    )

    trainer.train()
    return model

### 4.2.1 企业知识问答系统

**系统架构**：

```
用户问题 → 问题处理 → 检索模块 → 知识库匹配
                              ↓
                        生成模块 → 答案生成 → 用户响应
```

**完整实现**：

```python
import hashlib
from typing import List, Dict, Optional
from dataclasses import dataclass

@dataclass
class Document:
    """文档结构"""
    id: str
    content: str
    metadata: Dict

class EnterpriseQASystem:
    """企业知识问答系统"""
    def __init__(self, config):
        self.config = config

        # 文档处理
        self.text_splitter = CharacterTextSplitter(
            chunk_size=config.chunk_size,
            chunk_overlap=config.chunk_overlap
        )

        # 嵌入模型
        self.embeddings = HuggingFaceEmbeddings(
            model_name=config.embedding_model
        )

        # 向量存储
        self.vectorstore = None

        # LLM
        self.llm = self._init_llm(config)

        # 缓存
        self.cache = {}

    def _init_llm(self, config):
        """初始化LLM"""
        if config.provider == "openai":
            return OpenAI(model=config.model_name, temperature=0)
        elif config.provider == "huggingface":
            return HuggingFaceHub(repo_id=config.model_name)
        else:
            raise ValueError(f"Unknown provider: {config.provider}")

    def index_documents(self, documents: List[Document]):
        """索引文档"""
        # 分割文档
        texts = []
        metadatas = []

        for doc in documents:
            chunks = self.text_splitter.split_text(doc.content)
            texts.extend(chunks)
            metadatas.extend([doc.metadata] * len(chunks))

        # 创建向量索引
        self.vectorstore = FAISS.from_texts(
            texts,
            self.embeddings,
            metadatas=metadatas
        )

        print(f"已索引 {len(texts)} 个文本块")

    def retrieve(self, query: str, top_k: int = 5) -> List[Document]:
        """检索相关文档"""
        docs = self.vectorstore.similarity_search(query, k=top_k)
        return docs

    def generate_answer(self, query: str, context_docs: List[Document]) -> str:
        """生成答案"""
        # 构建上下文
        context = "\n\n".join([
            f"[文档{i+1}]\n{doc.page_content}"
            for i, doc in enumerate(context_docs)
        ])

        # 构建提示
        prompt = f"""基于以下参考资料回答问题。如果资料中没有相关信息，请说明"我没有找到相关内容"。

参考资料：
{context}

问题：{query}

回答："""

        # 生成
        answer = self.llm(prompt)

        return answer

    def answer(self, question: str) -> Dict:
        """回答问题"""
        # 检查缓存
        cache_key = hashlib.md5(question.encode()).hexdigest()
        if cache_key in self.cache:
            return self.cache[cache_key]

        # 检索
        docs = self.retrieve(question)

        # 生成
        answer = self.generate_answer(question, docs)

        # 缓存结果
        result = {
            'answer': answer,
            'sources': [doc.metadata for doc in docs],
            'retrieved_chunks': len(docs)
        }

        self.cache[cache_key] = result

        return result

    def add_document(self, content: str, metadata: Dict):
        """添加新文档"""
        doc = Document(
            id=hashlib.md5(content.encode()).hexdigest(),
            content=content,
            metadata=metadata
        )

        # 添加到向量存储
        if self.vectorstore is None:
            self.index_documents([doc])
        else:
            self.vectorstore.add_texts(
                [doc.content],
                metadatas=[doc.metadata]
            )

# 使用示例
config = QAConfig(
    chunk_size=500,
    chunk_overlap=100,
    embedding_model="sentence-transformers/paraphrase-multilingual-MiniLM-L12-v2",
    provider="huggingface",
    model_name="meta-llama/Llama-2-7b-chat-hf"
)

qa_system = EnterpriseQASystem(config)

# 索引文档
documents = [
    Document(
        id="1",
        content="公司成立于2020年，专注于人工智能技术...",
        metadata={"source": "公司介绍", "department": "HR"}
    ),
    # ... 更多文档
]
qa_system.index_documents(documents)

# 问答
result = qa_system.answer("公司的成立时间是什么时候？")
print(f"答案: {result['answer']}")
```

### 4.2.3 客服机器人

```python
from enum import Enum
from typing import Dict, List

class Intent(Enum):
    """意图类型"""
    GREETING = "greeting"
    PRODUCT_INQUIRY = "product_inquiry"
    ORDER_STATUS = "order_status"
    COMPLAINT = "complaint"
    GOODBYE = "goodbye"
    UNKNOWN = "unknown"

class CustomerServiceBot:
    """客服机器人"""
    def __init__(self, llm_client, knowledge_base):
        self.llm = llm_client
        self.knowledge_base = knowledge_base
        self.conversation_history = []

    def detect_intent(self, user_message: str) -> Intent:
        """检测用户意图"""
        prompt = f"""请判断用户消息的意图类型，只能选择以下一种：
- greeting：问候
- product_inquiry：产品咨询
- order_status：订单状态查询
- complaint：投诉
- goodbye：告别
- unknown：未知

用户消息：{user_message}

意图类型："""

        intent_str = self.llm.generate(prompt, max_tokens=20).strip()

        try:
            return Intent(intent_str)
        except:
            return Intent.UNKNOWN

    def handle_greeting(self) -> str:
        """处理问候"""
        return "您好！我是智能客服，很高兴为您服务。请问有什么可以帮助您的？"

    def handle_product_inquiry(self, message: str) -> str:
        """处理产品咨询"""
        # 检索相关产品信息
        docs = self.knowledge_base.search(message, top_k=3)

        if docs:
            context = "\n".join([doc.content for doc in docs])
            prompt = f"""基于以下产品信息回答用户问题：

产品信息：
{context}

用户问题：{message}

回答："""

            return self.llm.generate(prompt, max_tokens=300)
        else:
            return "抱歉，我没有找到相关信息。请问您能提供更多细节吗？"

    def handle_complaint(self, message: str) -> str:
        """处理投诉"""
        prompt = f"""用户提出了投诉，请以专业、友善的方式回应：

用户投诉：{message}

回复要求：
- 表示理解和歉意
- 说明将尽快处理
- 提供解决方案或联系渠道

回复："""

        return self.llm.generate(prompt, max_tokens=200)

    def chat(self, user_message: str) -> str:
        """对话入口"""
        # 检测意图
        intent = self.detect_intent(user_message)

        # 处理不同意图
        if intent == Intent.GREETING:
            response = self.handle_greeting()
        elif intent == Intent.PRODUCT_INQUIRY:
            response = self.handle_product_inquiry(user_message)
        elif intent == Intent.COMPLAINT:
            response = self.handle_complaint(user_message)
        elif intent == Intent.GOODBYE:
            response = "感谢您的咨询，祝您生活愉快！再见！"
        else:
            response = "抱歉，我没有理解您的意思。请您换个方式描述您的问题。"

        # 更新对话历史
        self.conversation_history.append({
            'user': user_message,
            'assistant': response,
            'intent': intent.value
        })

        return response

# 使用示例
bot = CustomerServiceBot(llm_client, knowledge_base)

while True:
    user_input = input("用户: ")
    if user_input.lower() in ['退出', '再见', 'bye']:
        print("客服: 再见！")
        break

    response = bot.chat(user_input)
    print(f"客服: {response}")
```

def check_gpu_memory():
    """检查GPU显存"""
    import torch
    if torch.cuda.is_available():
        for i in range(torch.cuda.device_count()):
            total = torch.cuda.get_device_properties(i).total_memory / 1024**3

import time
import psutil
import torch

class PerformanceProfiler:
    """性能分析器"""
    def __init__(self):
        self.measurements = []

    def profile_generation(self, model, prompt, num_runs=10):
        """分析生成性能"""
        tokenizer = model.__class__.from_pretrained("meta-llama/Llama-2-7b-chat-hf")

        results = {
            'preprocessing_time': [],
            'inference_time': [],
            'postprocessing_time': [],
            'total_time': [],
            'tokens_per_second': []
        }

        for _ in range(num_runs):
            # 预处理
            start = time.time()
            inputs = tokenizer(prompt, return_tensors="pt")
            preprocess_time = time.time() - start
            results['preprocessing_time'].append(preprocess_time)

            # 推理
            start = time.time()
            outputs = model.generate(**inputs, max_new_tokens=100)
            inference_time = time.time() - start
            results['inference_time'].append(inference_time)

            # 后处理
            start = time.time()
            response = tokenizer.decode(outputs[0])
            postprocess_time = time.time() - start
            results['postprocessing_time'].append(postprocess_time)

            # 总计
            total_time = preprocess_time + inference_time + postprocess_time
            results['total_time'].append(total_time)

            num_tokens = len(outputs[0])
            results['tokens_per_second'].append(num_tokens / inference_time)

        # 计算统计
        stats = {}
        for key, values in results.items():
            stats[key] = {
                'mean': sum(values) / len(values),
                'min': min(values),
                'max': max(values)
            }

        return stats

    def profile_memory(self, model):
        """分析内存使用"""
        torch.cuda.empty_cache()

        memory_info = {
            'allocated': torch.cuda.memory_allocated() / 1024**3,  # GB
            'reserved': torch.cuda.memory_reserved() / 1024**3,
            'max_allocated': torch.cuda.max_memory_allocated() / 1024**3
        }

        return memory_info

    def system_resources(self):
        """获取系统资源使用"""
        return {
            'cpu_percent': psutil.cpu_percent(),
            'memory_percent': psutil.virtual_memory().percent,
            'disk_usage': psutil.disk_usage('/').percent
        }

```python
class AccuracyEvaluator:
    """精度评估器"""
    def __init__(self, model, tokenizer):
        self.model = model
        self.tokenizer = tokenizer

    def evaluate_generation(self, test_cases):
        """评估生成质量"""
        results = []

        for case in test_cases:
            prompt = case['prompt']
            expected = case['expected']
            reference = case.get('reference', None)

            # 生成
            inputs = self.tokenizer(prompt, return_tensors="pt")
            outputs = self.model.generate(**inputs, max_new_tokens=256)
            generated = self.tokenizer.decode(outputs[0], skip_special_tokens=True)

            # 计算指标
            metrics = {
                'exact_match': self._exact_match(generated, expected),
                'contains_key': self._contains_key(generated, expected),
                'perplexity': self._calculate_perplexity(prompt + generated)
            }

            results.append({
                'prompt': prompt,
                'expected': expected,
                'generated': generated,
                'metrics': metrics
            })

        return results

    def _exact_match(self, generated, expected):
        """精确匹配"""
        return generated.strip() == expected.strip()

    def _contains_key(self, generated, expected):
        """关键信息匹配"""
        # 简单关键词匹配
        expected_keywords = set(expected.lower().split())
        generated_words = set(generated.lower().split())

        overlap = len(expected_keywords & generated_words)
        return overlap / len(expected_keywords) if expected_keywords else 0

    def _calculate_perplexity(self, text):
        """计算困惑度（简化版）"""
        inputs = self.tokenizer(text, return_tensors="pt")
        with torch.no_grad():
            outputs = self.model(**inputs, labels=inputs['input_ids'])
            loss = outputs.loss
            perplexity = torch.exp(loss).item()
        return perplexity

    def generate_report(self, results):
        """生成评估报告"""
        total = len(results)
        exact_matches = sum(1 for r in results if r['metrics']['exact_match'])
        avg_key_match = sum(r['metrics']['contains_key'] for r in results) / total

        print("=== 生成质量评估报告 ===")
        print(f"总测试数: {total}")
        print(f"精确匹配: {exact_matches} ({exact_matches/total:.1%})")
        print(f"平均关键词匹配: {avg_key_match:.1%}")

        # 列出失败案例
        print("\n失败案例:")
        for r in results:
            if not r['metrics']['exact_match']:
                print(f"  - Prompt: {r['prompt'][:50]}...")
                print(f"    Expected: {r['expected'][:50]}...")
                print(f"    Generated: {r['generated'][:50]}...")
```
