```python
from transformers import AutoModelForCausalLM, AutoTokenizer

class TextGenerator:
    """文本生成器"""
    def __init__(self, model_name="meta-llama/Llama-2-7b-chat-hf"):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForCausalLM.from_pretrained(
            model_name,
            device_map="auto",
            torch_dtype=torch.float16
        )

    def generate(self, prompt, max_new_tokens=256, temperature=0.7):
        """生成文本"""
        inputs = self.tokenizer(prompt, return_tensors="pt").to(self.model.device)

        outputs = self.model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            temperature=temperature,
            do_sample=True,
            top_p=0.9
        )

        response = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        return response

# 使用示例
generator = TextGenerator()
```

```python
from typing import Iterator

class StreamingTextGenerator:
    """流式文本生成器"""
    def __init__(self, model_name):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForCausalLM.from_pretrained(
            model_name,
            device_map="auto",
            torch_dtype=torch.float16
        )

    def generate_stream(self, prompt, max_new_tokens=256) -> Iterator[str]:
        """流式生成文本"""
        inputs = self.tokenizer(prompt, return_tensors="pt").to(self.model.device)

        # 使用generate的stream模式
        for token in self.model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            do_sample=True,
            temperature=0.7
        ):
            yield self.tokenizer.decode(token, skip_special_tokens=True)

# 使用示例
```

```python
from transformers import AutoModelForSequenceClassification, AutoTokenizer
import torch

class TextClassifier:
    """文本分类器"""
    def __init__(self, model_name, num_labels=2):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(
            model_name,
            num_labels=num_labels
        )

    def classify(self, text, return_probs=False):
        """分类预测"""
        inputs = self.tokenizer(text, return_tensors="pt", padding=True, truncation=True)

        with torch.no_grad():
            outputs = self.model(**inputs)
            logits = outputs.logits

        probs = torch.softmax(logits, dim=-1)
        predicted_class = torch.argmax(probs, dim=-1).item()

        if return_probs:
            return {
                'class': predicted_class,
                'probabilities': probs[0].tolist()
            }
        return {'class': predicted_class}

    def train(self, train_texts, train_labels, val_texts, val_labels, epochs=3):
        """微调训练"""
        train_encodings = self.tokenizer(train_texts, truncation=True, padding=True, max_length=512)
        val_encodings = self.tokenizer(val_texts, truncation=True, padding=True, max_length=512)

        train_dataset = TextClassificationDataset(train_encodings, train_labels)
        val_dataset = TextClassificationDataset(val_encodings, val_labels)

        trainer = Trainer(
            model=self.model,
            train_dataset=train_dataset,
            eval_dataset=val_dataset,
            args=TrainingArguments(
                output_dir="./classifier",
                num_train_epochs=epochs,
                per_device_train_batch_size=16,
                evaluation_strategy="epoch"
            )
        )

        trainer.train()
        return self.model

class TextClassificationDataset(torch.utils.data.Dataset):
    def __init__(self, encodings, labels):
        self.encodings = encodings
        self.labels = labels

    def __len__(self):
        return len(self.labels)

    def __getitem__(self, idx):
        item = {key: torch.tensor(val[idx]) for key, val in self.encodings.items()}
        item['labels'] = torch.tensor(self.labels[idx])
        return item
```

```python
class MultiLabelClassifier:
    """多标签分类器"""
    def __init__(self, model_name, num_labels):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(
            model_name,
            num_labels=num_labels,
            problem_type="multi_label_classification"
        )

    def predict(self, text, threshold=0.5):
        """多标签预测"""
        inputs = self.tokenizer(text, return_tensors="pt", padding=True)

        with torch.no_grad():
            outputs = self.model(**inputs)
            probs = torch.sigmoid(outputs.logits)

        # 返回概率大于阈值的标签
        predicted_labels = (probs > threshold).nonzero(as_tuple=True)[1].tolist()
        label_probs = {i: probs[0][i].item() for i in predicted_labels}

        return {
            'labels': predicted_labels,
            'probabilities': label_probs
        }
```

### 3.1.3 情感分析应用

```python
class SentimentAnalyzer:
    """情感分析器"""
    def __init__(self, model_name="uer/roberta-base-finetuned-jd-binary-chinese"):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(model_name)
        self.labels = {0: "negative", 1: "positive"}

    def analyze(self, text, return_scores=False):
        """情感分析"""
        inputs = self.tokenizer(text, return_tensors="pt", padding=True, truncation=True)

        with torch.no_grad():
            outputs = self.model(**inputs)
            scores = outputs.logits
            probs = torch.softmax(scores, dim=-1)

        predicted_sentiment = self.labels[torch.argmax(probs, dim=-1).item()]

        if return_scores:
            return {
                'sentiment': predicted_sentiment,
                'positive_score': probs[0][1].item(),
                'negative_score': probs[0][0].item()
            }
        return {'sentiment': predicted_sentiment}

    def batch_analyze(self, texts, batch_size=32):
        """批量情感分析"""
        results = []
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i+batch_size]
            inputs = self.tokenizer(batch, return_tensors="pt", padding=True, truncation=True)

            with torch.no_grad():
                outputs = self.model(**inputs)
                probs = torch.softmax(outputs.logits, dim=-1)

            batch_results = [
                {'sentiment': self.labels[idx], 'score': probs[i][idx].item()}
                for i, idx in enumerate(torch.argmax(probs, dim=-1))
            ]
            results.extend(batch_results)

        return results

# 使用示例
analyzer = SentimentAnalyzer()

print(f"情感: {result['sentiment']}, 正面分数: {result['positive_score']:.4f}")
```

```python
from sentence_transformers import SentenceTransformer
import faiss
import numpy as np

class RAGQuestionAnswering:
    """检索增强问答系统"""
    def __init__(self, model_name="sentence-transformers/paraphrase-multilingual-MiniLM-L12-v2"):
        # 编码器
        self.encoder = SentenceTransformer(model_name)

        # 向量数据库
        self.index = None
        self.documents = []

    def build_index(self, documents, batch_size=32):
        """构建文档索引"""
        self.documents = documents

        # 编码文档
        embeddings = self.encoder.encode(documents, batch_size=batch_size, show_progress_bar=True)

        # 创建FAISS索引
        dimension = embeddings.shape[1]
        self.index = faiss.IndexFlatL2(dimension)
        self.index.add(np.array(embeddings))

    def retrieve(self, query, top_k=5):
        """检索相关文档"""
        # 编码查询
        query_embedding = self.encoder.encode([query])

        # 搜索
        distances, indices = self.index.search(np.array(query_embedding), top_k)

        results = []
        for i, (dist, idx) in enumerate(zip(distances[0], indices[0])):
            results.append({
                'document': self.documents[idx],
                'distance': float(dist),
                'rank': i + 1
            })

        return results

    def answer(self, question, context_docs=None, max_context_length=2048):
        """基于检索内容回答问题"""
        if context_docs is None:
            context_docs = self.retrieve(question, top_k=5)

        # 构造prompt
        context = "\n\n".join([f"[文档{i+1}] {doc['document']}" for i, doc in enumerate(context_docs)])

        prompt = f"""基于以下参考资料回答问题。如果资料中没有相关信息，请说明"我没有找到相关内容"。

参考资料：
{context}

问题：{question}

回答："""

        # 调用LLM生成答案
        response = self.llm.generate(prompt, max_new_tokens=512)

        return {
            'answer': response,
            'retrieved_documents': context_docs
        }
```

```python
class ExtractiveQA:
    """抽取式问答"""
    def __init__(self, model_name="bert-base-chinese"):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForQuestionAnswering.from_pretrained(model_name)

    def answer(self, question, context):
        """从上下文中抽取答案"""
        inputs = self.tokenizer(
            question,
            context,
            return_tensors="pt",
            max_length=512,
            truncation=True
        )

        with torch.no_grad():
            outputs = self.model(**inputs)

        start_scores = outputs.start_logits
        end_scores = outputs.end_logits

        # 找到最佳答案span
        start_idx = torch.argmax(start_scores)
        end_idx = torch.argmax(end_scores)

        # 确保end_idx >= start_idx
        if end_idx < start_idx:
            end_idx = start_idx

        # 解码答案
        answer_tokens = inputs['input_ids'][0][start_idx:end_idx+1]
        answer = self.tokenizer.decode(answer_tokens, skip_special_tokens=True)

        return {
            'answer': answer,
            'start_idx': start_idx.item(),
            'end_idx': end_idx.item(),
            'confidence': (start_scores[0][start_idx] + end_scores[0][end_idx]).item()
        }
```

```python
    """Few-shot提示构造"""
    def __init__(self, examples, template=None):
        self.examples = examples
        self.template = template or self._default_template()

    def _default_template(self, example):
        """默认模板：输入->输出"""
```

**多样本选择策略**：

```python
class DiverseFewShot:
    """多样本选择策略"""
    def __init__(self, embeddings_model):
        self.embeddings_model = embeddings_model

    def select_diverse_examples(self, query, candidates, k=5):
        """选择多样化的示例"""
        # 计算查询与候选的相似度
        query_emb = self.embeddings_model.encode([query])

        # 最大边际相关（MMR）选择
        selected = []
        remaining = candidates.copy()

        for _ in range(k):
            best_score = -float('inf')
            best_example = None

            for ex in remaining:
                ex_emb = self.embeddings_model.encode([ex['input']])

                # 相似度分数
                relevance = np.dot(query_emb, ex_emb.T).item()

                # 多样性分数（与已选示例的距离）
                if selected:
                    diversity = min(
                        1 - np.dot(ex_emb, self.embeddings_model.encode([s['input']]).T).item()
                        for s in selected
                    )
                else:
                    diversity = 0

                # MMR分数
                mmr_score = 0.5 * relevance + 0.5 * diversity

                if mmr_score > best_score:
                    best_score = mmr_score
                    best_example = ex

            if best_example:
                selected.append(best_example)
                remaining.remove(best_example)

        return selected
```

```python
class LLMAPIClient:

            "messages": [{"role": "user", "content": prompt}],

            "temperature": temperature,
            **kwargs

        async with aiohttp.ClientSession() as session:
            async with session.post(
                f"{self.base_url}/chat/completions",
                headers=headers,
                json=payload
            ) as response:
                result = await response.json()
                return result['choices'][0]['message']['content']

    async def batch_generate(
        self,
        prompts: List[str],
        max_concurrent: int = 5,
        **kwargs
    ) -> List[str]:
        """批量异步生成"""
        semaphore = asyncio.Semaphore(max_concurrent)

        async def generate_with_limit(prompt):
            async with semaphore:
                return await self.generate(prompt, **kwargs)

        tasks = [generate_with_limit(p) for p in prompts]
        return await asyncio.gather(*tasks)
```

### 3.3.2 错误处理

```python
class LLMError(Exception):
    """LLM相关错误基类"""
    pass

class RateLimitError(LLMError):
    """速率限制错误"""
    pass

class APIError(LLMError):
    """API调用错误"""
    pass

class TimeoutError(LLMError):
    """超时错误"""
    pass

class LLMClientWithErrorHandling:
    """带错误处理的LLM客户端"""
    def __init__(self, base_client):
        self.client = base_client
        self.max_retries = 3
        self.retry_delay = 1.0

    async def generate_with_retry(self, prompt, **kwargs):
        """带重试的生成"""
        last_error = None

        for attempt in range(self.max_retries):
            try:
                return await self.client.generate(prompt, **kwargs)

            except RateLimitError:
                # 指数退避
                wait_time = self.retry_delay * (2 ** attempt)
                print(f"速率限制，等待 {wait_time} 秒后重试...")
                await asyncio.sleep(wait_time)

            except TimeoutError:
                # 增加超时时间重试
                kwargs['timeout'] = kwargs.get('timeout', 30) * 2
                print(f"超时，增加超时时间后重试...")

            except APIError as e:
                if e.status_code >= 500:
                    # 服务器错误，重试
                    await asyncio.sleep(self.retry_delay)
                else:
                    # 客户端错误，不重试
                    raise

            except Exception as e:
                last_error = e

        raise last_error or LLMError("最大重试次数已用完")

    def generate_sync(self, prompt, **kwargs):
        """同步生成"""
        return asyncio.run(self.generate_with_retry(prompt, **kwargs))
```

### 3.3.3 重试机制

```python
import time
from functools import wraps

def retry_with_exponential_backoff(
    max_retries=5,
    initial_delay=1,
    max_delay=60,
    exponential_base=2
):
    """指数退避重试装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            delay = initial_delay
            last_exception = None

            for attempt in range(max_retries):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    last_exception = e

                    if attempt < max_retries - 1:
                        wait_time = min(delay * (exponential_base ** attempt), max_delay)
                        print(f"尝试 {attempt + 1} 失败，{wait_time}秒后重试...")
                        await asyncio.sleep(wait_time)
                    else:
                        print(f"所有重试尝试失败")

            raise last_exception

        return wrapper
    return decorator

class ResilientLLMClient:
    """有韧性的LLM客户端"""
    def __init__(self, base_url, api_key):
        self.client = LLMAPIClient(base_url, api_key)

    @retry_with_exponential_backoff(max_retries=4, initial_delay=1)
    async def generate(self, prompt, **kwargs):
        return await self.client.generate(prompt, **kwargs)
```

### 3.3.4 超时控制

```python
import signal

class TimeoutException(Exception):
    pass

def timeout_handler(signum, frame):
    raise TimeoutException("Generation timed out")

class LLMDurationControl:
    """LLM生成时长控制"""
    def __init__(self, timeout_seconds=60):
        self.timeout = timeout_seconds

    async def generate_with_timeout(self, prompt, **kwargs):
        """带超时控制的生成"""
        try:
            # 设置信号处理器
            signal.signal(signal.SIGALRM, timeout_handler)
            signal.alarm(self.timeout)

            result = await self.client.generate(prompt, **kwargs)

            # 取消信号
            signal.alarm(0)

            return result

        except TimeoutException:
            raise TimeoutException(f"生成超时（{self.timeout}秒）")

    def generate_stream_with_timelimit(self, prompt, max_duration=60):
        """流式生成带时长限制"""
        start_time = time.time()
        elapsed = 0

        for token in self.client.generate_stream(prompt):
            elapsed = time.time() - start_time

            if elapsed >= max_duration:
                yield "[生成已超时]"
                break

            yield token

            # 可选：限制生成token数量
            if elapsed > max_duration:
                break
```

### 3.4.3 性能测试

```python
import time
import statistics

class PerformanceTest:
    """性能测试"""
    def __init__(self, client):
        self.client = client

    def test_latency(self, prompt, num_runs=100):
        """延迟测试"""
        latencies = []

        for _ in range(num_runs):
            start = time.time()
            self.client.generate(prompt)
            latency = (time.time() - start) * 1000  # ms
            latencies.append(latency)

        return {
            'mean': statistics.mean(latencies),
            'median': statistics.median(latencies),
            'p95': sorted(latencies)[int(len(latencies) * 0.95)],
            'p99': sorted(latencies)[int(len(latencies) * 0.99)],
            'std': statistics.stdev(latencies)
        }

    def test_throughput(self, prompt, duration_seconds=60):
        """吞吐量测试"""
        count = 0
        start = time.time()

        while time.time() - start < duration_seconds:
            self.client.generate(prompt)
            count += 1

        return {
            'requests_per_second': count / duration_seconds
        }

    def test_concurrent_throughput(self, prompts, num_concurrent=10):
        """并发吞吐量测试"""
        start = time.time()

        async def run_concurrent():
            tasks = [self.client.generate_async(p) for p in prompts]
            await asyncio.gather(*tasks)

        asyncio.run(run_concurrent())

        elapsed = time.time() - start

        return {
            'total_requests': len(prompts),
            'elapsed_seconds': elapsed,
            'requests_per_second': len(prompts) / elapsed
        }

    def run_all_tests(self, test_prompts):
        """运行所有性能测试"""
        results = {}

        results['latency'] = self.test_latency(test_prompts[0])
        results['throughput'] = self.test_throughput(test_prompts[0], duration_seconds=30)
        results['concurrent'] = self.test_concurrent_throughput(test_prompts)

        return results
```

```python
import logging
from datetime import datetime
from prometheus_client import Counter, Histogram, Gauge

# 指标定义
REQUEST_COUNT = Counter('llm_requests_total', 'Total requests', ['status'])
REQUEST_LATENCY = Histogram('llm_request_latency_seconds', 'Request latency')
ACTIVE_REQUESTS = Gauge('llm_active_requests', 'Active requests')
MODEL_LOAD_TIME = Histogram('llm_model_load_seconds', 'Model load time')

class MonitoringLogger:
    """监控日志器"""
    def __init__(self, logger_name="llm_service"):
        self.logger = logging.getLogger(logger_name)
        self.logger.setLevel(logging.INFO)

        # 文件处理器
        fh = logging.FileHandler(f"llm_{datetime.now().strftime('%Y%m%d')}.log")
        fh.setLevel(logging.INFO)

        # 控制台处理器
        ch = logging.StreamHandler()
        ch.setLevel(logging.INFO)

        formatter = logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        )
        fh.setFormatter(formatter)
        ch.setFormatter(formatter)

        self.logger.addHandler(fh)
        self.logger.addHandler(ch)

    def log_request(self, prompt_length, response_length, latency):
        """记录请求日志"""
        REQUEST_LATENCY.observe(latency)
        REQUEST_COUNT.labels(status='success').inc()

        self.logger.info(
            f"Request completed - "
            f"prompt_len: {prompt_length}, "
            f"response_len: {response_length}, "
            f"latency: {latency:.3f}s"
        )

    def log_error(self, error, context=None):
        """记录错误日志"""
        REQUEST_COUNT.labels(status='error').inc()

        self.logger.error(
            f"Request failed - error: {str(error)}, "
            f"context: {context or {}}"
        )

    def log_model_load(self, duration):
        """记录模型加载时间"""
        MODEL_LOAD_TIME.observe(duration)
        self.logger.info(f"Model loaded in {duration:.2f}s")

# 使用示例
monitor = MonitoringLogger()

# 在API调用中

    start = time.time()
    response = generator.generate(prompt)
    latency = time.time() - start

    monitor.log_request(
        prompt_length=len(prompt),
        response_length=len(response),
        latency=latency
    )
except Exception as e:
    monitor.log_error(e, {'prompt_length': len(prompt)})
```
