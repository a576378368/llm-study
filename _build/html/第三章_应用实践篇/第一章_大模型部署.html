1.1.3 性能评估指标#

核心指标：

  1. 延迟（Latency）：

    • 首Token延迟（Time to First Token，TTFT）

    • 逐Token生成延迟

    • 端到端延迟

  2. 吞吐量（Throughput）：

    • 每秒请求数（QPS）

    • 每秒Token数（Tokens/s）

  3. 资源利用率：

    • GPU利用率

    • 显存占用

    • CPU利用率

class PerformanceBenchmark:
    """性能基准测试"""
    def __init__(self, model, tokenizer):
        self.model = model
        self.tokenizer = tokenizer

    def benchmark_latency(self, prompt, num_runs=100):
        """延迟测试"""
        import time

        times = []
        for _ in range(num_runs):
            start = time.time()

            # 编码
            inputs = self.tokenizer(prompt, return_tensors='pt')

            # 生成
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=100,
                do_sample=False
            )

            # 解码
            response = self.tokenizer.decode(outputs[0])

            end = time.time()
            times.append(end - start)

        return {
            'mean': np.mean(times),
            'std': np.std(times),
            'p50': np.percentile(times, 50),
            'p95': np.percentile(times, 95),
            'p99': np.percentile(times, 99)
        }

    def benchmark_throughput(self, prompt, duration=60):
        """吞吐量测试"""
        import time

        start_time = time.time()
        num_requests = 0
        num_tokens = 0

        while time.time() - start_time < duration:
            inputs = self.tokenizer(prompt, return_tensors='pt')
            outputs = self.model.generate(**inputs, max_new_tokens=50)

            num_requests += 1
            num_tokens += outputs.shape[1]

        elapsed = time.time() - start_time

        return {
            'requests_per_second': num_requests / elapsed,
            'tokens_per_second': num_tokens / elapsed
        }

    def benchmark_concurrent(self, prompt, num_concurrent=10, num_requests=100):
        """并发测试"""
        import asyncio

        async def single_request():
            inputs = self.tokenizer(prompt, return_tensors='pt')
            outputs = await asyncio.create_task(
                self.model.generate_async(**inputs, max_new_tokens=50)
            )
            return outputs

        # 并发执行
        start = time.time()
        for _ in range(num_requests):
            asyncio.run(single_request())
        end = time.time()

        return {
            'total_time': end - start,
            'avg_latency': (end - start) / num_requests
        }

1.2.2 多GPU并行策略#

张量并行（Tensor Parallelism）：

将单一层的权重分割到多个GPU上。

import torch
import torch.nn as nn
from transformers import AutoModelForCausalLM

class TensorParallelModel:
    """张量并行模型"""
    def __init__(self, model_name, num_gpus=2):
        self.num_gpus = num_gpus
        self.model = self._load_tensor_parallel_model(model_name)

    def _load_tensor_parallel_model(self, model_name):
        """加载张量并行模型"""
        # 使用DeepSpeed或Megatron加载
        # 这里使用Hugging Face的accelerate库示例
        from accelerate import init_empty_weights, load_checkpoint_and_dispatch

        model = AutoModelForCausalLM.from_pretrained(
            model_name,
            device_map="auto",
            torch_dtype=torch.float16
        )
        return model

    def generate(self, prompt):
        inputs = self.tokenizer(prompt, return_tensors='pt')
        outputs = self.model.generate(**inputs)
        return self.tokenizer.decode(outputs[0])

流水线并行（Pipeline Parallelism）：

将模型的不同层分配到不同GPU上。

class PipelineParallelModel:
    """流水线并行模型"""
    def __init__(self, model, num_stages=4):
        self.num_stages = num_stages
        self.stages = self._split_model(model, num_stages)

    def _split_model(self, model, num_stages):
        """将模型切分为多个阶段"""
        layers = list(model.children())
        chunk_size = len(layers) // num_stages

        stages = []
        for i in range(num_stages):
            start = i * chunk_size
            end = start + chunk_size if i < num_stages - 1 else len(layers)
            stage = nn.Sequential(*layers[start:end])
            stages.append(stage.to(f'cuda:{i}'))

        return stages

    def forward(self, x):
        """流水线前向传播"""
        for i, stage in enumerate(self.stages):
            if i == 0:
                x = x.to('cuda:0')
            x = stage(x)
            if i < self.num_stages - 1:
                x = x.to(f'cuda:{i+1}')
        return x

import tensorrt as trt

def build_engine(model_path, fp16=True):
    """为Jetson构建TensorRT引擎"""
    logger = trt.Logger(trt.Logger.WARNING)

    with trt.Builder(logger) as builder:
        config = builder.create_builder_config()
        config.set_memory_pool_limit(trt.MemoryPoolType.WORKSPACE, 1 << 30)  # 1GB

        if fp16:
            config.set_flag(trt.BuilderFlag.FP16)

        network = builder.create_network(1 << int(trt.NetworkDefinitionCreationFlag.EXPLICIT_BATCH))
        parser = trt.OnnxParser(network, logger)

        with open(model_path, 'rb') as f:
            parser.parse(f.read())

        engine = builder.build_serialized_network(network, config)
        return engine

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Optional, List

app = FastAPI(title="LLM API", version="1.0")

class GenerateRequest(BaseModel):
    prompt: str
    max_tokens: Optional[int] = 256
    temperature: Optional[float] = 0.7
    top_p: Optional[float] = 0.9
    stop: Optional[List[str]] = None

class GenerateResponse(BaseModel):
    text: str
    finish_reason: str
    prompt_tokens: int
    completion_tokens: int

@app.post("/v1/completions", response_model=GenerateResponse)
async def generate(request: GenerateRequest):
    """文本补全接口"""
    try:
        inputs = tokenizer(request.prompt, return_tensors='pt').to(device)

        outputs = model.generate(
            **inputs,
            max_new_tokens=request.max_tokens,
            temperature=request.temperature,
            top_p=request.top_p,
            do_sample=True
        )

        response = tokenizer.decode(outputs[0], skip_special_tokens=True)

        return GenerateResponse(
            text=response,
            finish_reason="stop",
            prompt_tokens=len(inputs['input_ids'][0]),
            completion_tokens=len(outputs[0]) - len(inputs['input_ids'][0])
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

# 实现OpenAI API兼容接口
class OpenAICompatibility:
    """OpenAI API兼容层"""

    def __init__(self, model):
        self.model = model
        self.tokenizer = model.tokenizer

    def completions(self, prompt, **kwargs):
        """/v1/completions 兼容"""
        inputs = self.tokenizer(prompt, return_tensors='pt')
        outputs = self.model.generate(**inputs, **kwargs)
        return {
            "choices": [{
                "text": self.tokenizer.decode(outputs[0]),
                "finish_reason": "stop"
            }]
        }

    def embeddings(self, texts):
        """/v1/embeddings 兼容"""
        embeddings = []
        for text in texts:
            inputs = self.tokenizer(text, return_tensors='pt')
            outputs = self.model(**inputs)
            embedding = outputs.last_hidden_state[:, 0].cpu().numpy()
            embeddings.append(embedding.tolist())
        return {"data": [{"embedding": e} for e in embeddings]}

1.3.2 批处理优化#

动态批处理：

class DynamicBatcher:
    """动态批处理器"""
    def __init__(self, model, max_batch_size=32, max_wait_time_ms=100):
        self.model = model
        self.max_batch_size = max_batch_size
        self.max_wait_time = max_wait_time_ms / 1000.0  # 转换为秒

        self.pending_requests = []
        self.lock = asyncio.Lock()

    async def add_request(self, prompt, params):
        """添加请求到批处理队列"""
        future = asyncio.Future()
        self.pending_requests.append({
            'prompt': prompt,
            'params': params,
            'future': future
        })

        # 检查是否可以执行
        if len(self.pending_requests) >= self.max_batch_size:
            await self._execute_batch()

        return await future

    async def _execute_batch(self):
        """执行一批请求"""
        async with self.lock:
            if len(self.pending_requests) == 0:
                return

            batch = self.pending_requests[:self.max_batch_size]
            self.pending_requests = self.pending_requests[self.max_batch_size:]

        # 准备batch输入
        prompts = [r['prompt'] for r in batch]
        params = batch[0]['params']

        # 执行推理
        outputs = self.model.generate(prompts, **params)

        # 完成futures
        for i, request in enumerate(batch):
            request['future'].set_result(outputs[i])

KV Cache优化：

class KVCacheManager:
    """KV缓存管理器"""
    def __init__(self, max_cache_size=10000):
        self.cache = {}
        self.max_size = max_cache_size
        self.hits = 0
        self.misses = 0

    def get(self, prompt_hash):
        """获取缓存的KV"""
        if prompt_hash in self.cache:
            self.hits += 1
            return self.cache[prompt_hash]
        self.misses += 1
        return None

    def set(self, prompt_hash, kv_cache):
        """设置KV缓存"""
        if len(self.cache) >= self.max_size:
            # LRU淘汰
            oldest_key = next(iter(self.cache))
            del self.cache[oldest_key]

        self.cache[prompt_hash] = kv_cache

    def get_hit_rate(self):
        """获取缓存命中率"""
        total = self.hits + self.misses
        return self.hits / total if total > 0 else 0

class SemanticCache:
    """语义缓存 - 基于向量相似度"""
    def __init__(self, embedding_model, threshold=0.95):
        self.embedding_model = embedding_model
        self.threshold = threshold
        self.cache = []
        self.embeddings = []

    def get(self, prompt):
        """查找相似缓存"""
        query_emb = self.embedding_model.encode(prompt)

        for i, cached_emb in enumerate(self.embeddings):
            similarity = cosine_similarity(query_emb, cached_emb)
            if similarity > self.threshold:
                return self.cache[i]['response']

        return None

    def set(self, prompt, response):
        """添加缓存"""
        embedding = self.embedding_model.encode(prompt)
        self.cache.append({'prompt': prompt, 'response': response})
        self.embeddings.append(embedding)

1.3.4 负载均衡#

负载均衡策略：

import hashlib
from collections import defaultdict

class LoadBalancer:
    """请求负载均衡器"""
    def __init__(self, backends):
        """
        backends: [{"url": "http://gpu1:8000", "weight": 1}, ...]
        """
        self.backends = backends
        self.connection_counts = defaultdict(int)

    def select_backend(self, strategy='least_connections'):
        """选择后端"""
        if strategy == 'least_connections':
            # 最少连接
            min_count = min(self.connection_counts.values())
            candidates = [b for b in self.backends
                         if self.connection_counts[b['url']] == min_count]
            return candidates[0] if candidates else self.backends[0]

        elif strategy == 'round_robin':
            # 轮询
            return self.backends[0]  # 简化实现

        elif strategy == 'hash':
            # 哈希（相同prompt到相同后端）
            return self.backends[0]  # 简化实现

        return self.backends[0]

    async def forward(self, prompt, params):
        """转发请求"""
        backend = self.select_backend('least_connections')
        self.connection_counts[backend['url']] += 1

        try:
            response = await call_backend(backend['url'], prompt, params)
            return response
        finally:
            self.connection_counts[backend['url']] -= 1

Prometheus监控指标：

from prometheus_client import Counter, Histogram, Gauge, start_http_server

# 定义指标
REQUEST_COUNT = Counter(
    'llm_requests_total',
    'Total number of requests',
    ['model', 'status']
)

REQUEST_LATENCY = Histogram(
    'llm_request_duration_seconds',
    'Request latency',
    ['model']
)

GPU_UTILIZATION = Gauge(
    'llm_gpu_utilization',
    'GPU utilization percentage',
    ['gpu_id']
)

ACTIVE_REQUESTS = Gauge(
    'llm_active_requests',
    'Number of active requests',
    []
)

def track_request_metrics(model_name):
    """请求指标跟踪装饰器"""
    def decorator(func):
        async def wrapper(*args, **kwargs):
            ACTIVE_REQUESTS.inc()
            start_time = time.time()

            try:
                result = await func(*args, **kwargs)
                REQUEST_COUNT.labels(model=model_name, status='success').inc()
                return result
            except Exception as e:
                REQUEST_COUNT.labels(model=model_name, status='error').inc()
                raise
            finally:
                duration = time.time() - start_time
                REQUEST_LATENCY.labels(model=model_name).observe(duration)
                ACTIVE_REQUESTS.dec()

        return wrapper
    return decorator

日志配置：

import logging
import json
from datetime import datetime

class StructuredLogger:
    """结构化日志记录器"""
    def __init__(self, name='llm-service'):
        self.logger = logging.getLogger(name)
        self.logger.setLevel(logging.INFO)

        # 控制台输出
        console_handler = logging.StreamHandler()
        console_handler.setFormatter(logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        ))
        self.logger.addHandler(console_handler)

    def log_request(self, prompt, response, latency, metadata=None):
        """记录请求日志"""
        log_data = {
            'timestamp': datetime.utcnow().isoformat(),
            'type': 'request',
            'prompt_length': len(prompt),
            'response_length': len(response),
            'latency_ms': latency,
            **(metadata or {})
        }
        self.logger.info(json.dumps(log_data))

    def log_error(self, error, context=None):
        """记录错误日志"""
        log_data = {
            'timestamp': datetime.utcnow().isoformat(),
            'type': 'error',
            'error': str(error),
            **(context or {})
        }
        self.logger.error(json.dumps(log_data))

    def log_metrics(self, metrics):
        """记录指标日志"""
        log_data = {
            'timestamp': datetime.utcnow().isoformat(),
            'type': 'metrics',
            **metrics
        }
        self.logger.info(json.dumps(log_data))
